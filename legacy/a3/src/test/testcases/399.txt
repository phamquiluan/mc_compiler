    int bintogray(int bin){
      int a, b, result, i;
      result=0;i=0;
      do{
        a = bin % 10;
        bin = bin / 10;
        b = bin % 10;
        if ((a && !b) || (!a && b)){
          result = result + pow(10, i);
        }
        i=i+1;
      }while (bin != 0);
      return result;
